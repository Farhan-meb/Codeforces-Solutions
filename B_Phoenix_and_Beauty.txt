int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        int n,k;
        cin>>n>>k;
        vector<int>fs;
        int N[n+10];
        int f=0;
        map<int,int>mp;
        for(int i=1; i<=n; i++)
        {
            cin>>N[i];
            if(!mp[N[i]])
            {
                fs.pb(N[i]);
                mp[N[i]] = 1;
                f++;
            }
        }

        if(n==k)
        {
            cout<<n<<endl;
            for(int i=1; i<=n; i++)
                cout<<N[i]<<" ";
            cout<<endl;
            continue;
        }

        if(f>k)
        {
            cout<<-1<<endl;
            continue;
        }
        if(f<k){
            for(int i=1;i<=n;i++){
                if(!mp[i]){
                    f++;
                    fs.pb(i);
                }
                if(f==k) break;
            }
        }

        vector<int>v;

        for(int i=1; i<=n; i++)
        {
            for(int i=0; i<fs.size(); i++)
            {
                v.pb(fs[i]);

            }
        }






        cout<<v.size()<<endl;

        for(int i=0; i<v.size(); i++)
            cout<<v[i]<<" ";
        cout<<endl;

    }


}
