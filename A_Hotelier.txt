vector<int>v;
void convertBin(int num)
{
    while(num>0)
    {
        v.pb(num%2);
        num/=2;

    }
    reverse(v.begin(),v.end());
}
int convertArrayToNum(vector<int>v)
{
    int res = v[0];
    for(int i=1; i<8; i++)
    {
        res*=10;
        res+=v[i];
    }
    return res;
}
